    volume_spike: { multiplier: 3, notify: true }   // 成交量3倍
  },
  institutional: {
    foreign_3d_buy: { days: 3, notify: true },      // 外資連3日買
    margin_surge: { increase_pct: 20, notify: true } // 融資暴增20%

  }
};
